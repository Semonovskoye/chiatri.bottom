function T(v){
    return (v ?? "").toString().replace(/<[^>]*>/g,"").trim();
}

function P(n){
    return String(n).padStart(2,"0");
}

function vnDate(){
    const d = new Date();
    return `${P(d.getDate())}/${P(d.getMonth()+1)}/${d.getFullYear()}`;
}

function vnTime(){
    const d = new Date();
    return `${P(d.getHours())}:${P(d.getMinutes())}`;
}

        localStorage.studentName ||
        window.state?.studentInfo?.name ||
        "Chu Hoàng Lâm"
    );
}

function typeVN(t){
    return {
        "multiple-choice":"Trắc nghiệm",
        "matching":"Ghép nối",
        "fill-blank":"Điền khuyết",
        "true-false":"Đúng/Sai",
        "ordering":"Sắp xếp",
        "image-recognition":"Nhận diện hình ảnh"
    }[t] || t;
}

function levelVN(x){
    return {
        easy:"Dễ",
        medium:"Trung bình",
        hard:"Khó"
    }[x] || x;
}

function choiceText(x){
    if(typeof x === "string") return x.replace(/^[A-D]\.\s*/,"");

    if(typeof x === "object"){
        return T(x.text || x.label || x.content || x.value);
    }

    return T(x);
}

function getQuestion(q){
    return T(q.question || q.title || q.content);
}

function getAnswer(q,type){

    if(type === "multiple-choice" || type === "image-recognition"){

        const i = q.correctIndex ?? q.correct ?? 0;

        return choiceText(q.options?.[i]);
    }

    if(type === "true-false"){

        const v = q.answer ?? q.correct;

        return (v === true || v === 1 || v === "1")
            ? "Đúng"
            : "Sai";
    }

    if(type === "fill-blank"){

        return T(q.answer || q.correct);
    }

    if(type === "matching"){

        return (q.pairs || [])
            .map(x => `${T(x.left)} - ${T(x.right)}`)
            .join("; ");
    }

    if(type === "ordering"){

        return (q.correct || [])
            .map(i => T(q.items?.[i]))
            .join(" → ");
    }

    return T(q.answer || q.correct);
}
